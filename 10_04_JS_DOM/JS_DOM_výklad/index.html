Changing DOM in JS

textContent


            const headerElm = document.querySelector('header');
            headerElm

            // <div class="header">Page header</div>
            headerElm.textContent = 'New header'
            headerElm
            // <div class="header">New header</div>

Styling


            const headerElm = document.querySelector('.header');
            headerElm.style.color = 'white'
            headerElm.style.paddingTop = '2rem'

Changing class


            const headerElm = document.querySelector('header');
            headerElm
            // <div class="header">Page header</div>

            headerElm.className = 'new-header'
            headerElm
            // <div class="new-header">Page header</div>

Adding or removing class


            const btnElm = document.querySelector('.btn');
            btnElm
            // <button class="btn">Click me</button>

            btnElm.classList.add = 'btn-primary'
            btnElm
            // <div class="btn btn-primary">Click me</div>

            btnElm.classList.remove = 'btn-primary'
            btnElm
            // <div class="btn">Click me</div>

            const petElement = document.querySelector('.pet');
